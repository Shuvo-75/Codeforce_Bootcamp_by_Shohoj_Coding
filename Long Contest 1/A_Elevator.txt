// Problem link: https://codeforces.com/problemset/problem/120/a

#include <bits/stdc++.h>

using namespace std;

int main()
{
    string side;
    int floor_number; 
    cin >> side >> floor_number;

    if(side == "front")
    {
        if(floor_number % 2 == 0)
        {
            cout << "R";
        }else
        {
            cout << "L"; 
        }
    }
    
    if(side == "back")
    {
        if(floor_number % 2 == 0)
        {
            cout << "L";
        }else
        {
            cout << "R"; 
        }
    }
    return 0;
}
